### স্তর 3: রিজলভার রেপুটেশন সিস্টেম

```solidity
// রিজলভার স্টেক স্লাশ + রিগ্রেশন স্কোর
struct ResolverProfile {
    uint256 reputationScore;      // 0-10000 (basis points)
    uint256 totalResolved;        // মোট সমাধান সংখ্যা
    uint256 correctResolutions;   // সঠিক সমাধান সংখ্যা
    uint256 stakedAmount;         // স্টেক করা PLKY টোকেন
    uint256 slashCount;           // কতবার স্লাশ হয়েছে
    bool    isActive;             // সক্রিয় কিনা
    uint256 joinTimestamp;        // যোগদানের সময়
}

// রিজলভার স্টেক স্টেক করতে হবে
function stakeAsResolver(uint256 amount) external {
    require(amount >= MIN_STAKE, "Min stake required");
    plkyToken.transferFrom(msg.sender, address(this), amount);
    resolvers[msg.sender].stakedAmount += amount;
    resolvers[msg.sender].isActive = true;
}

// ভুল সমাধানে স্টেক স্লাশ
function slashResolver(address resolver, uint256 amount, string calldata reason) external onlyArbiter {
    ResolverProfile storage r = resolvers[resolver];
    uint256 slashAmount = min(amount, r.stakedAmount);
    r.stakedAmount -= slashAmount;
    r.slashCount++;
    r.reputationScore = calculateReputation(r);
    
    // স্লাশ করা টোরন রিওয়ার্ড পুলে দিয়া হয়
    // সাফল ডিসপুটারেরদের কাছে যায়
}
```

**Reputation Algorithm:**
```
reputationScore = (correctResolutions / totalResolved) * 10000 * (1 - slashCount * 0.1)

Weight in voting = stakeAmount * (reputationScore / 10000)

Example:
- Resolver A: 90/100 correct, 0 slashes, score = 9000, weight = stake * 0.9
- Resolver B: 50/100 correct, 2 slashes, score = 4000, weight = stake * 0.4
```

### স্তর 4: অর্থনৈতিক সুরক্ষা (বন্ধ সিস্টেম)

| কাজ | বন্ধ লাগে | রিটার্ন |
|------|------------|--------|
| Verdict প্রস্তাব করা | 100 PLKY | সমাধান হলে রিটার্ন + রিওয়ার্ড |
| Dispute উত্থাপন | 50 PLKY | সাঠিক হলে রিটার্ন + রিওয়ার্ড |
| Evidence Submit | 10 PLKY | সাঠিক হলে রিটার্঩ (সাদরণ সময়য়) |
| Resolver Stake | 1000 PLKY | Slash হলে হারিয়ে |

**Economic Incentives:**
- সাঠিক সমাধান: Resolver রিওয়ার্ড + স্টেক রিটার্ন
- ভুল সমাধান: Resolver স্টেক স্লাশ + রিজার্ভেশন কাম
- সাঠিক ডিসপুট: রিওয়ার্ড সাঠিক disputers এর কাছে
- ভুল ডিসপুট: Disputer স্টেক স্লাশ
